Class PApplicative@{d c p} (T : Type@{d} → Type@{c}) :=
{ AppP : Type@{d} → Type@{p}
; ppure : ∀ {A : Type@{d}} {P : AppP A}, A → T A
; pap : ∀ {A B : Type@{d}} {P : AppP B}, T (A → B) → T A → T B
}.

Class Applicative@{d c} (T : Type@{d} → Type@{c}) :=
{ pure : ∀ {A : Type@{d}}, A → T A
; ap : ∀ {A B : Type@{d}}, T (A → B) → T A → T B
}.
